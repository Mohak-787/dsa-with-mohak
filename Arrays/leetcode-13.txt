/**
* - Solution in Python3
*/
class Solution:
    def romanToInt(self, s: str) -> int:
        convert = {
            'I':1,
            'V':5,
            'X':10,
            'L':50,
            'C':100,
            'D':500,
            'M':1000
        }
        
        output = 0
        for i in range(len(s)):
            if i < len(s) - 1 and convert[s[i]] < convert[s[i+1]]:
                output -= convert[s[i]]
            else:
                output += convert[s[i]]
        
        return output    

/**
* - Step-by-Step Algorithm
*/
Step 1: Create a dictionary that maps each Roman numeral character to its integer value (I=1, V=5, X=10, etc.).
Step 2: Initialize a variable output to 0. This will store the final result.
Step 3: Go through each character of the Roman numeral string from left to right.
Step 4: For each character, check if it is not the last character AND its value is less than the value of the next character.
Step 5: If the current value is smaller than the next value, subtract its value from output (this handles cases like IV = 4).
Step 6: Otherwise, add its value to output.
Step 7: Continue this process until all characters in the string are processed.
Step 8: Return the final value stored in output.
