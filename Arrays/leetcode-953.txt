/**
* - Solution in Python3
*/
class Solution:
    def isAlienSorted(self, words: List[str], order: str) -> bool:
        order_map = {c: i for i, c in enumerate(order)}
        
        for i in range(1, len(words)):
            prev, curr = words[i - 1], words[i]
            n = min(len(prev), len(curr))
        
            if prev[:n] == curr[:n] and len(prev) > len(curr):
                return False
            
            for j in range(n):
                if prev[j] != curr[j]:
                    if order_map[prev[j]] > order_map[curr[j]]:
                        return False
                    break 
        
        return True

/**
* - Step-by-Step Algorithm
*/
Step 1: Create a dictionary called order_map that stores each character of the alien alphabet and its position (index) in the given order string.
Step 2: Start looping through the list of words from the second word to the end.
Step 3: For each word, compare it with the previous word. Let the previous word be prev and the current word be curr.
Step 4: Find the minimum length between the two words to avoid index errors during comparison.
Step 5: Check if both words are identical up to the minimum length, but the previous word is longer than the current word. If yes, return False (because longer word should come after shorter prefix).
Step 6: Compare characters one by one from both words up to the minimum length.
Step 7: If a mismatch is found, compare their positions using order_map. If the character in prev has a higher order value than the character in curr, return False. Otherwise, break the loop and move to the next pair of words.
Step 8: Continue this process for all adjacent word pairs.
Step 9: If no violations are found, return True.
